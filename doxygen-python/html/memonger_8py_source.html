21 def share_grad_blobs(net, losses, param_grads, namescope):
22  '''
23  Implements similar optimization as Torch's shareGradInput():
24  for the gradients that are passed between layers, share blobs between
25  operators when possible. This yields significant memory savings with
26  deep networks.
27 
28  Returns an optimized protobuf (assign to net._net)
29  '''
30  def is_grad_blob(b):
31  name = str(b)
32  # Note: need to look at _{namescope} pattern as it matches
33  # to handle the auto-split gradients
34  return "_grad" in name and (name.startswith(namescope) or
35  name.startswith("_" + namescope)) and name not in param_grads
36 
37  def is_grad_op(op):
38  # TODO: something smarter
39  for inp in op.input:
40  if is_grad_blob(inp):
41  return True
42  for out in op.output:
43  if is_grad_blob(out):
44  return True
45  return False
46 
47  start_time = time.time()
48  log.warn("NOTE: Executing *experimental* memonger to " +
49  "optimize gradient memory")
50 
51  # Collect ops that have something to do with
52  # gradients
53  if not namescope.endswith("/"):
54  namescope += "/"
55 
56  netproto = copy.deepcopy(net.Proto())
57  grad_ops = [op for op in netproto.op if is_grad_op(op)]
58 
59  # Create mapping from blobs to ops
60  blobs_to_ops = collections.defaultdict(lambda: [])
61  blob_input_count = collections.defaultdict(lambda: 0)
62  op_inputs = collections.defaultdict(lambda: 0)
63  op_visit_count = collections.defaultdict(lambda: 0)
64  for i, op in enumerate(grad_ops):
65  for inp in op.input:
66  if is_grad_blob(inp) or inp in losses:
67  # Ignore in-place transformation ops (self cycles)
68  if inp not in op.output:
69  blobs_to_ops[inp].append(i)
70  op_inputs[i] += 1
71 
72  # Traverse operators starting from the loss blobs.
73  # Keep tabs on when blobs are seen first and last, and also
74  # when operators have their input satisfied. Share blobs only
75  # under same branch, avoiding problems with parallel workers.
76  output_blobs = set()
77  mapping = {}
78 
79  def descend(op_idx, free_blobs):
80  cur_op = grad_ops[op_idx]
81  new_free_blobs = set()
82  for inp in cur_op.input:
83  if is_grad_blob(inp):
84  blob_input_count[inp] += 1
85  if blob_input_count[inp] == len(blobs_to_ops[inp]):
86  actual_blob = inp if inp not in mapping else mapping[inp]
87  new_free_blobs.add(actual_blob)
88 
89  for outp in cur_op.output:
90  if is_grad_blob(outp):
91  if outp not in output_blobs:
92  # First seen this blob as output, can assign to a free blob
93  for freeb in free_blobs:
94  mapping[outp] = freeb
95  free_blobs.remove(freeb)
96  break
97 
98  output_blobs.add(outp)
99 
100  free_blobs.update(new_free_blobs)
101 
102  first_branch = True
103  for outp in cur_op.output:
104  for inp_op_idx in blobs_to_ops[outp]:
105  op_visit_count[inp_op_idx] += 1
106 
107  # Descend only if we have satisfied all inputs
108  if op_visit_count[inp_op_idx] == op_inputs[inp_op_idx]:
109  free_blobs_fwd = free_blobs if first_branch else set()
110  first_branch = False
111  descend(inp_op_idx, free_blobs_fwd)
112 
113  # Start DFS from the losses
114  for loss in losses:
115  for op_idx in blobs_to_ops[loss]:
116  descend(op_idx, set())
117 
118  # Rename the shared blobs
119  shared_blobs = set(mapping.values())
120  renamed = {}
121  for j, b in enumerate(shared_blobs):
122  renamed[b] = namescope + "__m{}_".format(j)
123 
124  # Final mapping
125  for k, v in mapping.items():
126  mapping[k] = renamed[v]
127 
128  # Add the originators
129  mapping.update(renamed)
130  log.info("Remapping {} blobs, using {} shared".format(
131  len(mapping), len(renamed),
132  ))
133  apply_assignments(netproto, mapping)
134  log.info("Gradient memory optimization took {} secs".format(
135  time.time() - start_time),
136  )
137  return netproto

144 def compute_ranges(linearized_ops):
145  blobs = collections.defaultdict(lambda: LiveRange(defined=None, used=None))
146  for i, op in enumerate(linearized_ops):
147  for blob in op.input:
148  used = blobs[blob].used
149  if used is None:
150  used = i
151  else:
152  used = max(used, i)
153  blobs[blob] = blobs[blob]._replace(used=used)
154  for blob in op.output:
155  defined = blobs[blob].defined
156  if defined is None:
157  defined = i
158  else:
159  defined = min(defined, i)
160  blobs[blob] = blobs[blob]._replace(defined=defined)
161 
162  return blobs

186 def compute_assignments(ranges, static_blobs):
187  # Sort the ranges based on when they are last used.
188  # If LiveRange.used is None, then the blob is never used and could
189  # be consumed externally. Sort these to the end of the list as opposed
190  # to the beginning so that they can be shared as well.
191  ranges = sorted(
192  list(ranges.items()),
193  key=lambda p: (p[1].used is None, p[1].used),
194  )
195  assignments = []
196  for (name, range_) in ranges:
197  assigned = False
198  for assignment in assignments:
199  if is_compatible(range_, assignment, static_blobs):
200  assignment.append((name, range_))
201  assigned = True
202  break
203  if assigned:
204  continue
205  assignments.append([(name, range_)])
206  return assignments

209 def compute_interference_graph(ops):
210  g = nx.DiGraph()
211  for i, op in enumerate(ops):
212  g.add_node(i, op=op)
213  for i, parent_op in enumerate(ops):
214  for j, child_op in enumerate(ops):
215  if i == j:
216  continue
217  if any(output in child_op.input for output in parent_op.output):
218  deps = set(child_op.input).intersection(parent_op.output)
219  g.add_edge(i, j, deps=deps)
220  assert nx.is_directed_acyclic_graph(g), child_op
221  return g

241 def optimize_interference(net, static_blobs,
242  ordering_function=topological_sort_traversal):
243  """
244  1) Use a BFS traversal of the execution graph to generate an
245  ordering of the node executions.
246  2) Generate use-def ranges for each `blob` in the BFS traversal
247  order.
248  3) Assign blobs to `canonical blobs`
249  4) Rename blobs to canonical blobs
250  """
251  net = copy.deepcopy(net)
252  g = compute_interference_graph(net.op)
253  ordering = ordering_function(g)
254  linearized_ops = [net.op[i] for i in ordering]
255 
256  # Reorder ops in net based on the computed linearlized order.
257  # If the graph has multiple topological orderings and if the NetDef's
258  # ordering differs from the order used to compute ranges, then the
259  # runtime might end up overwriting blobs before they are used.
260  del net.op[:]
261  net.op.extend(linearized_ops)
262 
263  ranges = compute_ranges(linearized_ops)
264  assignments = compute_assignments(ranges, static_blobs)
265  blob_assignments = compute_blob_assignments(assignments)
266  apply_assignments(net, blob_assignments)
267  return Optimization(
268  net=net,
269  blob_assignments=blob_assignments,
270  assignments=assignments)
