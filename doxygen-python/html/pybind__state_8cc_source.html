98 void switchWorkspaceInternal(const std::string& name, bool create_if_missing) {

141 py::object fetchBlob(Workspace* ws, const std::string& name) {
142  CAFFE_ENFORCE(ws->HasBlob(name), "Can't find blob: ", name);
143  const caffe2::Blob& blob = *(ws->GetBlob(name));
144  auto fetcher = CreateFetcher(blob.meta().id());
145  if (fetcher) {
146  return fetcher->Fetch(blob);
147  } else {
148  // If there is no fetcher registered, return a metainfo string.
149  // If all branches failed, we will return a metainfo string.
150  std::stringstream ss;
151  ss << caffe2::string(name) << ", a C++ native class of type "
152  << blob.TypeName() << ".";
153  return py::str(ss.str());
154  }
155 }

158 bool PythonOpBase::RunOnDevice() {
159  std::vector<TensorCPU*> inputs;
160  inputs.reserve(InputSize());
161  for (auto i = 0; i < InputSize(); ++i) {
162  inputs.push_back(const_cast<TensorCPU*>(&Input(i)));
163  }
164  std::vector<TensorCPU*> outputs;
165  outputs.reserve(OutputSize());
166  for (auto i = 0; i < OutputSize(); ++i) {
167  outputs.push_back(Output(i));
168  }
169  auto& pyFunc = getFunc();
170  {
171  // Acquire GIL for call to Python runtime.
172  py::gil_scoped_acquire g;
173  try {
174  if (pyFunc.needs_workspace) {
175  pyFunc.py_func(inputs, outputs, ws_);
176  } else {
177  pyFunc.py_func(inputs, outputs);
178  }
179  } catch (const py::error_already_set& e) {
180  LOG(ERROR) << "Exception encountered running PythonOp function: "
181  << e.what() << "\nTraceback: ";
182  PyObject *type = nullptr, *value = nullptr, *trace = nullptr;
183  PyErr_Fetch(&type, &value, &trace);
184  PyTracebackObject* traceback =
185  reinterpret_cast<PyTracebackObject*>(trace);
186  vector<PyTracebackObject*> trace_vec;
187  while (traceback) {
188  trace_vec.push_back(traceback);
189  traceback = traceback->tb_next;
190  }
191  for (int i = trace_vec.size() - 1; i >= 0; --i) {
192  int line = trace_vec[i]->tb_lineno;
193  const char* filename =
194  PyString_AsString(trace_vec[i]->tb_frame->f_code->co_filename);
195  const char* funcname =
196  PyString_AsString(trace_vec[i]->tb_frame->f_code->co_name);
197  LOG(ERROR) << " # " << trace_vec.size() - i - 1 << " " << filename
198  << " (" << line << "): " << funcname;
199  }
200  Py_XDECREF(type);
201  Py_XDECREF(value);
202  Py_XDECREF(trace);
203  return false;
204  }
205  }
206  return true;
207 }

252 static bool ParseProtobufFromLargeString(const string& str, Message* proto) {
253  ::google::protobuf::io::ArrayInputStream input_stream(str.data(), str.size());
254  ::google::protobuf::io::CodedInputStream coded_stream(&input_stream);
255  // Set PlanDef message size limit to 1G.
256  coded_stream.SetTotalBytesLimit(1024LL << 20, 512LL << 20);
257  return proto->ParseFromCodedStream(&coded_stream);
