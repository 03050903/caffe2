9 '''
10 This module provides a python-land multithreaded data input mechanism
11 for Caffe2 nets.
12 
13 Basic usage is as follows:
14  coordinator = data_workers.init_data_input_workers(
15  net,
16  ["data", "label"],
17  my_fetch_fun,
18  batch_size=32,
19  input_source_name="train"
20  )
21  ...
22  coordinator.start()
23 
24 First argument is the Caffe2 net (or model helper), and second argument
25 is list of input blobs that are to be fed.
26 
27 Argument 'input_source_name' is used to distinguish different sources of data,
28 such as train or test data. This is to ensure the data does not get mixed up,
29 although two nets would share blobs.
30 
31 To do the actual data loading, one defines a "fetcher function"
32 that has call signature
33  my_fetch_fun(worker_id, batch_size)
34 
35 This function returns a list of numpy arrays corresponding to the different
36 input blobs. In the example above, it would return two arrays, one for the
37 data blob and another for the labels. These arrays can have arbitrary number
38 of elements (i.e they do not need to match the batch size). The batch size
39 is provided for the function as a hint only.
40 
41 For example, fetcher function could download images from a remote service or
42 load random images from a directory on a file system.
43 
44 For a dummy example, see the data_workers_test unit test.
45 
46 Note that for data_parallel_models, init_data_input_workers will be called
47 for each GPU. Note that the 'coordinator' returned by the function is same
48 each time.
49 '''

158  print("Wait for workers to die")

171  def _get(self):
172  while self.is_active():
173  try:
174  return self._internal_queue.get(block=True, timeout=0.5)
175  except Queue.Empty:
176  continue

179  def put(self, chunk):
180  if len(chunk) == 0:
181  print("Worker provided zero length input")
182  return
183  while self.is_active():
184  try:
185  qsize = self._internal_queue.qsize()
186  if qsize < 2 and (time.time() - self._last_warning) > LOG_INT_SECS:
187  print("Warning, data loading lagging behind: " +
188  "name={}".format(qsize, self._input_source_name))
189  self._last_warning = time.time()
190  self._counter += 1
191  self._internal_queue.put(chunk, block=True, timeout=0.5)
192  self._log_inputs_per_interval(chunk[0].shape[0])
193  return
194  except Queue.Full:
195  log.debug("Queue full: stalling fetchers...")

198  def _enqueue_batch(self):
199  '''
200  This pulls data from the python-side queue and collects them
201  into batch-sized pieces.
202  '''
203  cur_batch = [np.array([]) for d in self._input_blob_names]
204 
205  # Collect data until we have a full batch size
206  while cur_batch[0].shape[0] < self._batch_size and self.is_active():
207  chunk = self._get()
208  if chunk is None:
209  continue
210 
211  for j, chunk_elem in enumerate(chunk):
212  if cur_batch[j].shape[0] == 0:
213  cur_batch[j] = chunk_elem.copy()
214  else:
215  cur_batch[j] = np.append(cur_batch[j], chunk_elem, axis=0)
216 
217  # Return data over the batch size back to queue
218  if cur_batch[0].shape[0] > self._batch_size:
219  leftover = [c[self._batch_size:] for c in cur_batch]
220  cur_batch = [c[:self._batch_size] for c in cur_batch]
221  try:
222  self._internal_queue.put(leftover, block=False)
223  except Queue.Full:
224  pass
225 
226  assert cur_batch[0].shape[0] == self._batch_size
227 
228  if self.is_active():
229  for b, q, c in zip(self._input_blob_names, self._queues, cur_batch):
230  self._enqueue(b, q, c)

232  def _enqueue(self, blob_name, queue, data_arr):
233  '''
234  Enqueue the correctly sized batch arrays to Caffe2's queue.
235  '''
236  scratch_name = self._namescope + blob_name + \

247  "SafeEnqueueBlobs",

332 def fetcher(coordinator, worker_id, fetch_fun, batch_size, input_blob_names):
333  while coordinator.is_active():
334  try:
335  input_data = fetch_fun(worker_id, batch_size)
336  if input_data is None:
337  print("Fetcher function returned None")
338  continue
339 
340  assert len(input_data) == len(input_blob_names), \
341  "Expecting data blob for each input"
342  for d in input_data:
343  assert isinstance(d, np.ndarray), \
344  "Fetcher function must return a numpy array"
345  for d in input_data[1:]:
346  assert d.shape[0] == input_data[0].shape[0], \
347  "Each returned input must have equal number of samples"
348 
349  coordinator.put(input_data)
350  except Exception as e:
351  logging.exception("Exception in fetcher", e)
352  coordinator._stop("Exception in fetcher {}: {}".format(
353  worker_id, e
