101 class SegmentsTester(TesterBase):
102  def split(self, data, segment_ids, indices=None):
103  """
104  Given:
105  data[M1 x M2 x ... x Md]
106  the input data
107  indices[N] the index of each entry of segment_ids into data,
108  where 0 <= index[i] < M1,
109  with default indices=[0,1,...N]
110  segment_ids[N] the segment_id for each entry of indices,
111 
112  returns K outputs, each one containing data entries corresponding
113  to one of the segments present in `segment_ids`.
114  """
115  if segment_ids.size == 0:
116  return []
117  K = max(segment_ids) + 1
118  outputs = [
119  np.zeros(
120  (np.count_nonzero(segment_ids == seg_id), ) + data.shape[1:],
121  dtype=data.dtype
122  ) for seg_id in range(0, K)
123  ]
124  counts = np.zeros(K, dtype=int)
125  for i, seg_id in enumerate(segment_ids):
126  data_idx = i if indices is None else indices[i]
127  outputs[seg_id][counts[seg_id]] = data[data_idx]
128  counts[seg_id] += 1
129  return outputs

131  def unsplit(self, extra_shape, inputs, segment_ids):
132  """ Inverse operation to `split`, with indices=None """
133  output = np.zeros((len(segment_ids), ) + extra_shape)
134  if len(segment_ids) == 0:
135  return output
136  K = max(segment_ids) + 1
137  counts = np.zeros(K, dtype=int)
138  for i, seg_id in enumerate(segment_ids):
139  output[i] = inputs[seg_id][counts[seg_id]]
140  counts[seg_id] += 1
141  return output

146  K = len(lengths)

215 def max_grad(grad_out, outputs, inputs):
216  flat_inputs = inputs[0].flatten()
217  flat_outputs = np.array(outputs[0]).flatten()
218  flat_grad_in = np.zeros(flat_inputs.shape)
219  flat_grad_out = np.array(grad_out).flatten()
220  blocks = inputs[0].shape[0]
221  block_size = flat_inputs.shape[0] // blocks
222 
223  for i in range(block_size):
224  out_grad = flat_grad_out[i]
225  out = flat_outputs[i]
226  for j in range(blocks):
227  idx = j * block_size + i
228  if out == flat_inputs[idx]:
229  flat_grad_in[idx] = out_grad
230  break
231 
232  return np.resize(flat_grad_in, inputs[0].shape)
