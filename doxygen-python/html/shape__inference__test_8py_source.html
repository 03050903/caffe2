318  def InferTensorRunAndCompare(self, model):
319  '''
320  Runs shape inference, and then the model to check
321  that the inferred shapes agree with the actual ones
322  '''
323  (shapes, types) = workspace.InferShapesAndTypes(
324  [model.param_init_net, model.net],
325  )
326 
327  # .. Create net
328  workspace.RunNetOnce(model.param_init_net)
329  workspace.CreateNet(model.net)
330  workspace.RunNet(model.Proto().name)
331 
332  # ... and then check the shapes mismatch
333  correct_shapes = {}
334  correct_types = {}
335  for b in workspace.Blobs():
336  arr = workspace.FetchBlob(b)
337  correct_shapes[b] = arr.shape
338  if type(arr) is np.ndarray:
339  if arr.dtype == np.dtype('float32'):
340  correct_types[b] = caffe2_pb2.TensorProto.FLOAT
341  elif arr.dtype == np.dtype('int32'):
342  correct_types[b] = caffe2_pb2.TensorProto.INT32
343  # BYTE
344  # STRING
345  elif arr.dtype == np.dtype('bool'):
346  correct_types[b] = caffe2_pb2.TensorProto.BOOL
347  elif arr.dtype == np.dtype('uint8'):
348  correct_types[b] = caffe2_pb2.TensorProto.UINT8
349  elif arr.dtype == np.dtype('int8'):
350  correct_types[b] = caffe2_pb2.TensorProto.INT8
351  elif arr.dtype == np.dtype('uint16'):
352  correct_types[b] = caffe2_pb2.TensorProto.UINT16
353  elif arr.dtype == np.dtype('int16'):
354  correct_types[b] = caffe2_pb2.TensorProto.INT16
355  elif arr.dtype == np.dtype('int64'):
356  correct_types[b] = caffe2_pb2.TensorProto.INT64
357  elif arr.dtype == np.dtype('float16'):
358  correct_types[b] = caffe2_pb2.TensorProto.FLOAT16
359  elif arr.dtype == np.dtype('float64'):
360  correct_types[b] = caffe2_pb2.TensorProto.DOUBLE
361  else:
362  correct_types[b] = "unknown {}".format(arr.dtype)
363  else:
364  correct_types[b] = str(type(arr))
365 
366  for b in correct_shapes:
367  self.assertTrue(
368  np.array_equal(
369  np.array(shapes[b]).astype(np.int32),
370  np.array(correct_shapes[b]).astype(np.int32)
371  ),
372  "Shape {} mismatch: {} vs. {}".format(
373  b, shapes[b], correct_shapes[b]
374  )
375  )
376  self.assertFalse(
377  b not in types and b in correct_types,
378  "Type for {} not defined".format(b),
379  )
380  self.assertEqual(
381  types[b],
382  correct_types[b],
383  "Type {} mismatch: {} vs. {}".format(
384  b, types[b], correct_types[b],
385  )
386  )
