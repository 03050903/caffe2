92  FetchedBlob FetchTensor(const Tensor<Context>& tensor, bool force_copy) {
93  FetchedBlob result;
94  CAFFE_ENFORCE_GE(tensor.size(), 0, "Trying to fetch unitilized tensor");
95  const int numpy_type = CaffeToNumpyType(tensor.meta());
96  CAFFE_ENFORCE(
97  numpy_type != -1,
98  "This tensor's data type is not supported: ",
99  tensor.meta().name(),
100  ".");
101  std::vector<npy_intp> npy_dims;
102  for (const auto dim : tensor.dims()) {
103  npy_dims.push_back(dim);
104  }
105  result.copied = force_copy || NeedsCopy(tensor.meta());
106  void* outPtr;
107  if (result.copied) {
108  result.obj = pybind11::object(
109  PyArray_SimpleNew(tensor.ndim(), npy_dims.data(), numpy_type),
110  /* borrowed */ false);
111  outPtr = static_cast<void*>(
112  PyArray_DATA(reinterpret_cast<PyArrayObject*>(result.obj.ptr())));
113  } else {
114  outPtr = const_cast<Tensor<Context>&>(tensor).raw_mutable_data();
115  result.obj = pybind11::object(
116  PyArray_SimpleNewFromData(
117  tensor.ndim(), npy_dims.data(), numpy_type, outPtr),
118  /* borrowed */ false);
119  }
120 
121  if (numpy_type == NPY_OBJECT) {
122  PyObject** outObj = reinterpret_cast<PyObject**>(outPtr);
123  auto* str = tensor.template data<std::string>();
124  for (int i = 0; i < tensor.size(); ++i) {
125  outObj[i] = PyBytes_FromStringAndSize(str->data(), str->size());
126  str++;
127  // cleanup on failure
128  if (outObj[i] == nullptr) {
129  for (int j = 0; j < i; ++j) {
130  Py_DECREF(outObj[j]);
131  }
132  CAFFE_THROW("Failed to allocate string for ndarray of strings.");
133  }
134  }
135  return result;
136  }
137 
138  if (result.copied) {
139  Context context;
140  context.template CopyBytes<Context, CPUContext>(
141  tensor.nbytes(), tensor.raw_data(), outPtr);
142  context.FinishDeviceComputation();
143  }
144  return result;
145  }

149 class TensorFeeder : public BlobFeederBase {
150  public:
151  void FeedTensor(
152  const DeviceOption& option,
153  PyArrayObject* original_array,
154  Tensor<Context>* tensor) {
155  PyArrayObject* array = PyArray_GETCONTIGUOUS(original_array);
156  auto g = MakeGuard([&]() { Py_XDECREF(array); });
157 
158  const auto npy_type = PyArray_TYPE(array);
159  const TypeMeta& meta = NumpyTypeToCaffe(npy_type);
160  CAFFE_ENFORCE(
161  meta.id() != 0,
162  "This numpy data type is not supported: ",
163  PyArray_TYPE(array),
164  ".");
165  Context context(option);
166  context.SwitchToDevice();
167  // numpy requires long int as its dims.
168  int ndim = PyArray_NDIM(array);
169  npy_intp* npy_dims = PyArray_DIMS(array);
170  std::vector<TIndex> dims;
171  for (int i = 0; i < ndim; ++i) {
172  dims.push_back(npy_dims[i]);
173  }
174  tensor->Resize(dims);
175 
176  // Now, copy the data to the tensor.
177  switch (npy_type) {
178  case NPY_OBJECT: {
179  PyObject** input = reinterpret_cast<PyObject**>(PyArray_DATA(array));
180  auto* outPtr = tensor->template mutable_data<std::string>();
181  for (int i = 0; i < tensor->size(); ++i) {
182  char* str;
183  Py_ssize_t strSize;
184  CAFFE_ENFORCE(
185  PyBytes_AsStringAndSize(input[i], &str, &strSize) != -1,
186  "Unsupported python object type passed into ndarray.");
187  outPtr[i] = std::string(str, strSize);
188  }
189  } break;
190  default:
191  context.template CopyBytes<CPUContext, Context>(
192  tensor->size() * meta.itemsize(),
193  static_cast<void*>(PyArray_DATA(array)),
194  tensor->raw_mutable_data(meta));
195  }
196  context.FinishDeviceComputation();
197  }
